class Solution {
    public boolean isSubsequence(String s, String t) 
    {
        int sLength = s.length();
        int tLength = t.length();
        
        if (tLength < sLength) 
        {
            return false;
        }
        
	// keep two index pointers
        int indexT = 0;
        int prevIndexT = 0;

        for (int i = 0; i < sLength; i++)
        {
            char letter = s.charAt(i);
            if (i == 0) // case: first letter
            {
                indexT = t.indexOf(letter);
            }
            else // case: the rest
            {
                prevIndexT = indexT;  // update previous index T
                indexT = t.indexOf(letter, prevIndexT +1); // only search string t from last index T to the end
            }
            if (indexT == -1) // if letter is not found, return falsel;
            {
                return false;
            }      
        }
        return true; // if manage to make it here, it means all letters in s have been found in t in correct order.
    }
}
