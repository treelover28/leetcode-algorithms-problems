class Solution {
    // brute-force solution
    public int[] twoSum(int[] nums, int target) 
    {
        for (int i = 0; i < nums.length; i++)
        {
            for (int j = i + 1; j < nums.length; j++) 
                // j always start ahead of i because can't use same number twice
            {
                if (nums[i] + nums[j] == target)
                {
                    return new int[] {i, j};
                }
            }
        }
        return null;
    }
}
