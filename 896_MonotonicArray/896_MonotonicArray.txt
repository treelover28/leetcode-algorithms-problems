class Solution 
{
    public boolean isMonotonic(int[] A) 
    {
        if (A.length==1) 
        {
            return true;
        }
        
        int n= A.length;
        boolean currentTrend = (A[n-1]-A[0]) > 0; // check trend by taking last value take away first value
        for (int i=0; i<n-1; i++)
        {
            if (A[i+1]!=A[i] && (A[i+1]-A[i]>0)!=currentTrend) 
            {
                // only check neighbours that are different
                // if the trend is broken, return false
                return false;
            }
        }
        return true;
    }
}
